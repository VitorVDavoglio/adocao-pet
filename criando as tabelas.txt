CREATE TABLE admins(
	cd_admin INT(11) AUTO_INCREMENT PRIMARY KEY,
	nome_admin VARCHAR(32) NOT NULL,
	e_mail_admin VARCHAR(64) NOT NULL,
	senha_admin VARCHAR(64) NOT NULL,
	ativ_admin TINYINT NOT NULL
);
